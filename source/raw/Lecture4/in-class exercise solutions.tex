{\Large \textbf{In-Class Exercise --- Solutions}}\\[6pt]
{\large \textbf{Lecture 4: Logistic Regression \& Support Vector Machines}}
\end{center}

\vspace{8pt}

Recall:
\[
s_i = y_i(w^T x_i + b),
\qquad
L(x_i,y_i,\theta)=\max(0,\;1-s_i).
\]

\vspace{6pt}
\noindent
\textbf{Given:}
\[
w=[0.6,-0.2],\qquad b=0.4
\]
\[
x_1=[2,1],\; y_1=+1
\qquad\qquad
x_2=[1,3],\; y_2=-1
\]

\vspace{10pt}
\noindent
\textbf{Solutions:}

\begin{enumerate}[label=(\arabic*), itemsep=10pt]

\item \textbf{Compute $s_i=y_i(w^T x_i+b)$}

For $(x_1,y_1)$:
\[
w^T x_1 = 0.6(2)+(-0.2)(1)=1.2-0.2=1.0
\]
\[
w^T x_1+b = 1.0+0.4=1.4
\]
\[
s_1 = (+1)(1.4)=1.4
\]
\[
\boxed{s_1=1.40}
\]

For $(x_2,y_2)$:
\[
w^T x_2=0.6(1)+(-0.2)(3)=0.6-0.6=0
\]
\[
w^T x_2+b=0+0.4=0.4
\]
\[
s_2 = (-1)(0.4)=-0.4
\]
\[
\boxed{s_2=-0.40}
\]

\item \textbf{Check the margin constraint: $s_i-1\ge 0$}

\[
s_1-1 = 1.4-1=0.4\ge 0 \Rightarrow \text{constraint satisfied}
\]
\[
\boxed{\text{$x_1$ satisfies margin constraint.}}
\]

\[
s_2-1 = -0.4-1=-1.4<0 \Rightarrow \text{constraint violated}
\]
\[
\boxed{\text{$x_2$ violates margin constraint.}}
\]

\item \textbf{Compute margin-perceptron cost}

\[
L_1=\max(0,1-s_1)=\max(0,1-1.4)=\max(0,-0.4)=0
\]
\[
\boxed{L_1=0.00}
\]

\[
L_2=\max(0,1-s_2)=\max(0,1-(-0.4))=\max(0,1.4)=1.4
\]
\[
\boxed{L_2=1.40}
\]

\end{enumerate}

\vspace{6pt}
\noindent
\textbf{Final answers (2 decimals):}
\[
\boxed{s_1=1.40,\; s_2=-0.40,\; L_1=0.00,\; L_2=1.40}
\]
